# Ambiente nodesjs

  $ mkdir nodesProjetos
  $ asdf list nodejs
  $ asdf local nodejs 12.18.2
  $ node -v

  $ yarn add mongoose
  $ yarn add @nestjs/mongoose 
  $ yarn add @types/mongoose --dev

  # Ambiente nestjs 

    $ npm i -g @nestjs/cli
    $ nest new apiagua

      $ cd api-jads

      $ yarn run start
      $ yarn run start:dev


  # Contruindo um To-Do

    $ nest g controller tasks
    $ nest g provider tasks/shared/task.service
    $ nest g module tasks
    $ nest g class tasks/shared/task
